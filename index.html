Examples

Models

Typesafe client representations of your models in an MVC-like environment. 

// Models.tst
$Classes(c => c.Name.EndsWith("Model"))[
class $Name {
    $Properties[
    public $name: $Type = $IsEnumerable[[]][$Default];
	]
	    
    constructor(json: any) {
        $Properties[
		this.$name = json.$name;
		]    
    }
}]


// EmployeeModel.cs
public class EmployeeModel
{
    public int Id { get; set; }
    public string Name { get; set; }
}


// EmployeeModel.ts (generated)
class EmployeeModel {
    public id:number = null;
    public name: string = null;
    
    constructor(json:any) {
        this.id = json.id;
        this.name = json.name;
    }             
}
